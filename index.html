-- Plan de cuentas CREATE TABLE cuentas ( id SERIAL PRIMARY KEY, codigo TEXT UNIQUE NOT NULL, nombre TEXT NOT NULL, tipo TEXT NOT NULL CHECK (tipo IN ('Activo','Pasivo','Patrimonio','Ingreso','Gasto')), created_at TIMESTAMPTZ DEFAULT NOW() ); -- Gastos CREATE TABLE gastos ( id SERIAL PRIMARY KEY, fecha DATE NOT NULL, descripcion TEXT NOT NULL, comprobante TEXT, cuenta_codigo TEXT REFERENCES cuentas(codigo), monto NUMERIC(14,2) NOT NULL, proyecto_id INTEGER REFERENCES proyectos(id), metodo TEXT, created_at TIMESTAMPTZ DEFAULT NOW() ); -- Asientos contables (cabecera) CREATE TABLE asientos ( id SERIAL PRIMARY KEY, fecha DATE NOT NULL, descripcion TEXT NOT NULL, referencia TEXT, origen TEXT DEFAULT 'manual', created_at TIMESTAMPTZ DEFAULT NOW() ); -- Líneas de asiento (detalle) CREATE TABLE asiento_lineas ( id SERIAL PRIMARY KEY, asiento_id INTEGER REFERENCES asientos(id) ON DELETE CASCADE, cuenta_codigo TEXT REFERENCES cuentas(codigo), detalle TEXT, debe NUMERIC(14,2) DEFAULT 0, haber NUMERIC(14,2) DEFAULT 0 ); -- Movimientos bancarios CREATE TABLE movimientos_banco ( id SERIAL PRIMARY KEY, fecha DATE NOT NULL, descripcion TEXT NOT NULL, monto NUMERIC(14,2) NOT NULL, conciliado BOOLEAN DEFAULT false, asiento_id INTEGER REFERENCES asientos(id), factura_nombre TEXT, factura_tipo TEXT, factura_data TEXT, created_at TIMESTAMPTZ DEFAULT NOW() );
